Auto Atribuição

var n = 3
n=n+4 simplificando -> n += 4
n=n-5 simplificando-> n -= 4
n=n*4 simplificando-> n *= 4
n=n/2 simplificando-> n /= 4
n=n**2 simplificando-> n **= 4
n=n%5 simplificando-> n %= 4
